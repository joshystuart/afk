    **1. Create `web/src/hooks/useFileTree.ts`:**
    React Query hook wrapping `workspaceApi.listDirectory`:
    ```typescript
    import { useQuery } from '@tanstack/react-query';
    import { workspaceApi } from '../api/workspace.api';
    import type { DirectoryListing } from '../api/types';

    export const useFileTree = (sessionId: string, path: string, enabled: boolean = true) => {
      return useQuery<DirectoryListing>({
        queryKey: ['workspace', 'files', sessionId, path],
        queryFn: () => workspaceApi.listDirectory(sessionId, path),
        enabled,
        staleTime: 30_000,
        retry: 1,
      });
    };
    ```

    **2. Create `web/src/hooks/useFileContent.ts`:**
    ```typescript
    import { useQuery } from '@tanstack/react-query';
    import { workspaceApi } from '../api/workspace.api';
    import type { FileContent } from '../api/types';

    export const useFileContent = (sessionId: string, path: string | null) => {
      return useQuery<FileContent>({
        queryKey: ['workspace', 'content', sessionId, path],
        queryFn: () => workspaceApi.getFileContent(sessionId, path!),
        enabled: !!path,
        staleTime: 30_000,
        retry: 1,
      });
    };
    ```

The OS handles protocol dispatch; AFK doesn't execute commands directly.                                             |
