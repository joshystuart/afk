**Create `web/src/hooks/useFileIndex.ts`:**

This hook fetches the flat file index for a session and provides a fuzzy search function via Fuse.js:

```typescript
import { useQuery, useQueryClient } from '@tanstack/react-query';
import { useEffect, useMemo, useCallback } from 'react';
import Fuse from 'fuse.js';
import { workspaceApi } from '../api/workspace.api';
import { useWebSocket } from './useWebSocket';

interface FileIndexEntry {
  path: string;
  name: string;       // last segment of path
  isDirectory: boolean; // paths ending in / are directories
}

export const useFileIndex = (sessionId: string | null) => {
  const queryClient = useQueryClient();
  const { socket } = useWebSocket();

  const query = useQuery({
    queryKey: ['workspace', 'index', sessionId],
    queryFn: () => workspaceApi.getFileIndex(sessionId!),
    enabled: !!sessionId,
    staleTime: 30_000,
    refetchOnWindowFocus: false,
  });

  // Invalidate file index when chat completes (files may have changed)
  useEffect(() => {
    if (!socket || !sessionId) return;
    const handleChatComplete = (payload: { sessionId: string }) => {
      if (payload.sessionId === sessionId) {
        queryClient.invalidateQueries({ queryKey: ['workspace', 'index', sessionId] });
      }
    };
    socket.on('chat.complete', handleChatComplete);
    return () => { socket.off('chat.complete', handleChatComplete); };
  }, [socket, sessionId, queryClient]);

  // Build Fuse index from file paths
  const entries: FileIndexEntry[] = useMemo(() => {
    if (!query.data?.files) return [];
    return query.data.files.map((filePath) => ({
      path: filePath,
      name: filePath.split('/').pop() || filePath,
      isDirectory: filePath.endsWith('/'),
    }));
  }, [query.data?.files]);

  const fuse = useMemo(() => {
    return new Fuse(entries, {
      threshold: 0.4,
      distance: 200,
      includeScore: true,
      keys: ['path', 'name'],
      shouldSort: true,
    });
  }, [entries]);

  const search = useCallback((filter: string): FileIndexEntry[] => {
    if (!filter) return entries.slice(0, 20);
    return fuse.search(filter).slice(0, 20).map((r) => r.item);
  }, [fuse, entries]);

  return { search, isLoading: query.isLoading, entries };
};
```

The `search` function returns up to 20 results. When filter is empty, it returns the first 20 entries (useful as initial suggestions per D-08). The entries include both files and folders per D-08. Per D-06 it searches across the full relative path with fuzzy matching. Per D-13, the index is invalidated when `chat.complete` fires for this session.
